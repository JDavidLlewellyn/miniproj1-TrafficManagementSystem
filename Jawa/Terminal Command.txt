    // ---------------- DB UTIL ----------------
    static class DBUtil {
        static Connection getConnection() throws SQLException {
            return DriverManager.getConnection(DBConfig.URL, DBConfig.USER, DBConfig.PASS);
        }

        // initialize database: create schema/tables
        static void initialize() {
            try (Connection conn = getConnection(); Statement st = conn.createStatement()) {
                // Create tables: vehicles, signals, violations
                String createVehicles = "CREATE TABLE IF NOT EXISTS vehicles ("
                        + "id INT AUTO_INCREMENT PRIMARY KEY,"
                        + "plate VARCHAR(20) NOT NULL UNIQUE,"
                        + "owner VARCHAR(100),"
                        + "type VARCHAR(50),"
                        + "registered_date DATE"
                        + ")";

                String createSignals = "CREATE TABLE IF NOT EXISTS signals ("
                        + "id INT AUTO_INCREMENT PRIMARY KEY,"
                        + "location VARCHAR(200) NOT NULL UNIQUE,"
                        + "status VARCHAR(20) NOT NULL," // e.g., OPERATIONAL, MAINTENANCE
                        + "last_inspection DATE"
                        + ")";

                String createViolations = "CREATE TABLE IF NOT EXISTS violations ("
                        + "id INT AUTO_INCREMENT PRIMARY KEY,"
                        + "vehicle_id INT NOT NULL,"
                        + "signal_id INT,"
                        + "violation_type VARCHAR(100),"
                        + "fine DECIMAL(10,2),"
                        + "violation_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,"
                        + "FOREIGN KEY (vehicle_id) REFERENCES vehicles(id) ON DELETE CASCADE,"
                        + "FOREIGN KEY (signal_id) REFERENCES signals(id) ON DELETE SET NULL"
                        + ")";

                st.execute(createVehicles);
                st.execute(createSignals);
                st.execute(createViolations);

            } catch (SQLException e) {
                JOptionPane.showMessageDialog(null, "DB Initialization Error: " + e.getMessage());
                e.printStackTrace();
            }
        }
    }

    private void addVehicle() {
        String plate = plateField.getText().trim();
        String owner = ownerField.getText().trim();
        String type = typeField.getText().trim();
        String regDate = regDateField.getText().trim();
        if (plate.isEmpty()) { JOptionPane.showMessageDialog(this, "Plate is required"); return; }

        String sql = "INSERT INTO vehicles (plate, owner, type, registered_date) VALUES (?, ?, ?, ?)";
        try (Connection conn = DBUtil.getConnection(); PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, plate);
            ps.setString(2, owner.isEmpty()? null : owner);
            ps.setString(3, type.isEmpty()? null : type);
            ps.setString(4, regDate.isEmpty()? null : regDate);
            ps.executeUpdate();
            JOptionPane.showMessageDialog(this, "Vehicle added.");
            clearVehicleForm();
            loadVehicles();
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error adding vehicle: " + e.getMessage());
            e.printStackTrace();
        }
    }

    private void clearVehicleForm() {
        plateField.setText(""); ownerField.setText(""); typeField.setText(""); regDateField.setText("");
    }

    private void loadVehicles() {
        vehicleModel.setRowCount(0);
        vehicleComboForViolation.removeAllItems();
        String sql = "SELECT id, plate, owner, type, registered_date FROM vehicles ORDER BY id";
        try (Connection conn = DBUtil.getConnection(); PreparedStatement ps = conn.prepareStatement(sql); ResultSet rs = ps.executeQuery()) {
            while (rs.next()) {
                Vector<Object> row = new Vector<>();
                int id = rs.getInt("id");
                String plate = rs.getString("plate");
                row.add(id); row.add(plate); row.add(rs.getString("owner")); row.add(rs.getString("type")); row.add(rs.getDate("registered_date"));
                vehicleModel.addRow(row);
                vehicleComboForViolation.addItem(id + ": " + plate);
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error loading vehicles: " + e.getMessage());
            e.printStackTrace();
        }
    }

    // ---------------- Violations ----------------
    private void addViolation() {
        String vsel = (String) vehicleComboForViolation.getSelectedItem();
        if (vsel == null) { JOptionPane.showMessageDialog(this, "No vehicle selected"); return; }
        int vehicleId = Integer.parseInt(vsel.split(":")[0].trim());
        String ssel = (String) signalComboForViolation.getSelectedItem();
        Integer signalId = null;
        if (ssel != null) signalId = Integer.parseInt(ssel.split(":")[0].trim());

        String vtype = violationTypeField.getText().trim();
        String fineStr = fineField.getText().trim();
        double fine = 0.0;
        try { if (!fineStr.isEmpty()) fine = Double.parseDouble(fineStr); } catch (NumberFormatException ex) { JOptionPane.showMessageDialog(this, "Invalid fine value"); return; }

        String sql = "INSERT INTO violations (vehicle_id, signal_id, violation_type, fine) VALUES (?, ?, ?, ?)";
        try (Connection conn = DBUtil.getConnection(); PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setInt(1, vehicleId);
            if (signalId == null) ps.setNull(2, Types.INTEGER); else ps.setInt(2, signalId);
            ps.setString(3, vtype.isEmpty()? "Unknown" : vtype);
            ps.setDouble(4, fine);
            ps.executeUpdate();
            JOptionPane.showMessageDialog(this, "Violation logged.");
            violationTypeField.setText(""); fineField.setText("");
            loadViolations();
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error logging violation: " + e.getMessage());
            e.printStackTrace();
        }
    }

    private void loadViolations() {
        violationModel.setRowCount(0);
        String sql = "SELECT v.id, ve.plate, s.location, v.violation_type, v.fine, v.violation_date "
                + "FROM violations v LEFT JOIN vehicles ve ON v.vehicle_id = ve.id LEFT JOIN signals s ON v.signal_id = s.id ORDER BY v.id";
        try (Connection conn = DBUtil.getConnection(); PreparedStatement ps = conn.prepareStatement(sql); ResultSet rs = ps.executeQuery()) {
            while (rs.next()) {
                Vector<Object> row = new Vector<>();
                row.add(rs.getInt(1));
                row.add(rs.getString(2));
                row.add(rs.getString(3));
                row.add(rs.getString(4));
                row.add(rs.getDouble(5));
                row.add(rs.getTimestamp(6));
                violationModel.addRow(row);
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Error loading violations: " + e.getMessage());
            e.printStackTrace();
        }
    }
